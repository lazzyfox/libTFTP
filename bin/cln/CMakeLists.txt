cmake_minimum_required(VERSION 3.9)
project(tftp_cln VERSION 0.0.4 DESCRIPTION "TFTP client test application build")
set(CMAKE_CXX_STANDARD 23)
set (CMAKE_CXX_COMPILER_VERSION 13)
set(CMAKE_C_COMPILER ${C_COMPILER})
set (CMAKE_CXX_COMPILER ${CXX_COMPILER})
message(STATUS "Making TFTP client")
set(CMAKE_CXX_FLAGS "-ggdb -pipe -std=c++2b -pthread -std=gnu++2b -Wall -Wextra -fPIC")
add_executable(tftp_cln ${PROJECT_SOURCE_DIR}/cln.cpp ${PROJECT_SOURCE_DIR}/../../src/libTFTP.hpp)
target_link_libraries(tftp_cln -lgcc -lstdc++ -ldl -lpthread)
